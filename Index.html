        function wrapLabel(label, maxLength = 16) {

            if (label.length <= maxLength) return label;

            const words = label.split(' ');

            const lines = [];

            let currentLine = '';

            words.forEach(word => {

                if ((currentLine + word).length > maxLength) {

                    if (currentLine.length > 0) lines.push(currentLine.trim());

                    currentLine = word + ' ';

                } else {

                    currentLine += word + ' ';

                }

            });

            if (currentLine.length > 0) lines.push(currentLine.trim());

            return lines;

        }
